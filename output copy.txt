from django.db import models


class PublishLog(models.Model):
    """
    Records every publish action (or attempt) made through Script Publisher.

    This table is the audit trail referenced in the security model:
    every deployment is recorded with who did it, what was changed,
    and where it was published.
    """

    STATUS_CHOICES = [
        ("success", "Success"),
        ("failed", "Failed"),
        ("pending_review", "Pending Review"),
    ]

    PUBLISH_METHOD_CHOICES = [
        ("botpassword", "BotPassword"),
        ("draft", "Draft / Notification only"),
    ]

    # Who published
    username = models.CharField(max_length=255, db_index=True)

    # What was published
    source_repo = models.URLField(max_length=1000)
    source_file = models.CharField(max_length=500)

    # Where it was published
    target_wiki = models.CharField(max_length=255)
    target_page = models.CharField(max_length=500)

    # Metadata provided at publish time
    version = models.CharField(max_length=100, blank=True)
    edit_summary = models.TextField(blank=True)

    # Outcome
    status = models.CharField(
        max_length=30,
        choices=STATUS_CHOICES,
        default="pending_review",
        db_index=True,
    )
    publish_method = models.CharField(
        max_length=30,
        choices=PUBLISH_METHOD_CHOICES,
        default="draft",
    )
    error_message = models.TextField(blank=True)

    # Timestamps
    created_at = models.DateTimeField(auto_now_add=True, db_index=True)

    class Meta:
        ordering = ["-created_at"]
        verbose_name = "Publish Log Entry"
        verbose_name_plural = "Publish Log Entries"

    def __str__(self):
        return f"{self.username} → {self.target_page} ({self.status})"


==== publisher\oauth.py ====
"""
oauth.py
--------
Handles the Wikimedia OAuth 2.0 flow for Script Publisher.

Scope: identity only.
  - We use OAuth to verify who the user is (username, groups, rights).
  - We do NOT use the OAuth token to edit wiki pages.
  - Editing is done either via the user's own BotPassword,
    or via a "notify + draft" flow where the user edits themselves.

This separation keeps the tool within its approved OAuth scope and
avoids the JS-editing security concerns raised in community review.

Flow:
  1. /get-started/ -> redirect to Wikimedia authorize URL
  2. Wikimedia -> /api/auth/callback/?code=...&state=...
  3. Exchange code for access token
  4. Fetch user profile (/oauth2/resource/profile)
  5. Store username + rights in Django session
  6. Redirect to /dashboard/
"""

import os
import secrets
import requests
from django.conf import settings


# ──────────────────────────────────────────────────────────────────────────────
# Constants (read from settings, which read from .env)
# ──────────────────────────────────────────────────────────────────────────────

CLIENT_ID = os.environ.get("OAUTH_CLIENT_ID", "")
CLIENT_SECRET = os.environ.get("OAUTH_CLIENT_SECRET", "")
REDIRECT_URI = os.environ.get("OAUTH_REDIRECT_URI", "")
AUTHORIZE_URL = os.environ.get(
    "OAUTH_AUTHORIZE_URL",
    "https://meta.wikimedia.org/w/rest.php/oauth2/authorize",
)
TOKEN_URL = os.environ.get(
    "OAUTH_TOKEN_URL",
    "https://meta.wikimedia.org/w/rest.php/oauth2/access_token",
)
PROFILE_URL = os.environ.get(
    "OAUTH_PROFILE_URL",
    "https://meta.wikimedia.org/w/rest.php/oauth2/resource/profile",
)

REQUEST_TIMEOUT = 10  # seconds


class OAuthError(Exception):
    """Raised when any step of the OAuth flow fails."""


# ──────────────────────────────────────────────────────────────────────────────
# Step 1 — Build the authorization redirect URL
# ──────────────────────────────────────────────────────────────────────────────

def build_authorize_url(request) -> str:
    """
    Generate a Wikimedia OAuth 2.0 authorization URL.

    A random `state` value is stored in the session to prevent CSRF attacks.
    The state is verified in handle_callback() before the code is exchanged.
    """
    state = secrets.token_urlsafe(32)
    request.session["oauth_state"] = state

    params = {
        "client_id": CLIENT_ID,
        "response_type": "code",
        "redirect_uri": REDIRECT_URI,
        "scope": "basic",  # 'basic' = read username + groups only
        "state": state,
    }

    from urllib.parse import urlencode
    return f"{AUTHORIZE_URL}?{urlencode(params)}"


# ──────────────────────────────────────────────────────────────────────────────
# Step 2 — Handle the callback, exchange code for token, fetch profile
# ──────────────────────────────────────────────────────────────────────────────

def handle_callback(request) -> dict:
    """
    Called from the OAuth callback view.

    Validates the state parameter, exchanges the authorization code for an
    access token, fetches the user profile, and returns a dict:

        {
            "username":  "ExampleUser",
            "groups":    ["*", "user", "autoconfirmed"],
            "rights":    ["read", "edit", ...],
            "token":     "<access token for profile reads only>"
        }

    Raises OAuthError if anything goes wrong.
    """
    # ── Validate state to prevent CSRF ──────────────────────────────────────
    returned_state = request.GET.get("state", "")
    stored_state = request.session.pop("oauth_state", None)

    if not stored_state or returned_state != stored_state:
        raise OAuthError(
            "State mismatch — possible CSRF attempt. Please try logging in again."
        )

    # ── Check for error response from Wikimedia ──────────────────────────────
    error = request.GET.get("error")
    if error:
        description = request.GET.get("error_description", error)
        raise OAuthError(f"Wikimedia denied authorization: {description}")

    code = request.GET.get("code")
    if not code:
        raise OAuthError("No authorization code received from Wikimedia.")

    # ── Exchange code for access token ───────────────────────────────────────
    token_data = _exchange_code(code)
    access_token = token_data.get("access_token")
    if not access_token:
        raise OAuthError("Token exchange succeeded but no access_token was returned.")

    # ── Fetch user profile ───────────────────────────────────────────────────
    profile = _fetch_profile(access_token)

    return {
        "username": profile.get("username", ""),
        "groups": profile.get("groups", []),
        "rights": profile.get("rights", []),
        "token": access_token,
    }


def _exchange_code(code: str) -> dict:
    """POST the authorization code to get an access token."""
    payload = {
        "grant_type": "authorization_code",
        "code": code,
        "redirect_uri": REDIRECT_URI,
        "client_id": CLIENT_ID,
        "client_secret": CLIENT_SECRET,
    }
    try:
        response = requests.post(
            TOKEN_URL,
            data=payload,
            headers={"User-Agent": _user_agent()},
            timeout=REQUEST_TIMEOUT,
        )
    except requests.exceptions.RequestException as e:
        raise OAuthError(f"Could not reach Wikimedia token endpoint: {e}")

    if not response.ok:
        raise OAuthError(
            f"Token exchange failed (HTTP {response.status_code}): {response.text}"
        )

    return response.json()


def _fetch_profile(access_token: str) -> dict:
    """Fetch the authenticated user's profile from Wikimedia."""
    try:
        response = requests.get(
            PROFILE_URL,
            headers={
                "Authorization": f"Bearer {access_token}",
                "User-Agent": _user_agent(),
            },
            timeout=REQUEST_TIMEOUT,
        )
    except requests.exceptions.RequestException as e:
        raise OAuthError(f"Could not reach Wikimedia profile endpoint: {e}")

    if not response.ok:
        raise OAuthError(
            f"Profile fetch failed (HTTP {response.status_code}): {response.text}"
        )

    return response.json()


# ──────────────────────────────────────────────────────────────────────────────
# Session helpers
# ──────────────────────────────────────────────────────────────────────────────

def login_user(request, profile: dict):
    """Store the authenticated user's info in the Django session."""
    request.session["user_username"] = profile["username"]
    request.session["user_groups"] = profile["groups"]
    request.session["user_rights"] = profile["rights"]
    request.session["user_token"] = profile["token"]
    # Preserve session across browser restarts
    request.session.set_expiry(86400 * 7)  # 7 days


def logout_user(request):
    """Clear all session data."""
    request.session.flush()


def get_current_user(request) -> dict | None:
    """
    Return the current user dict from session, or None if not logged in.

        {
            "username": "ExampleUser",
            "groups":   [...],
            "rights":   [...],
            "is_interface_admin": bool,
        }
    """
    username = request.session.get("user_username")
    if not username:
        return None

    groups = request.session.get("user_groups", [])
    rights = request.session.get("user_rights", [])

    return {
        "username": username,
        "groups": groups,
        "rights": rights,
        # Interface admins can edit MediaWiki: and *.js pages sitewide
        "is_interface_admin": "interface-admin" in groups,
    }


def require_login(view_func):
    """
    Decorator for views that require an authenticated user.
    Redirects to /get-started/ if not logged in.
    """
    from functools import wraps
    from django.shortcuts import redirect

    @wraps(view_func)
    def wrapper(request, *args, **kwargs):
        if not get_current_user(request):
            return redirect("get_started")
        return view_func(request, *args, **kwargs)

    return wrapper


# ──────────────────────────────────────────────────────────────────────────────
# Utilities
# ──────────────────────────────────────────────────────────────────────────────

def _user_agent() -> str:
    return "WikiScriptPublisher/1.0 (https://script-publisher.toolforge.org; Toolforge)"

==== publisher\repo_fetcher.py ====
"""
repo_fetcher.py
---------------
Fetches JS/CSS files directly from public GitHub or GitLab repositories
using their native REST APIs.

Replaces the previous gitingest.com dependency entirely.
No third-party services are called — all requests go directly to the
hosting platform's API. This keeps the tool self-contained and auditable.

Supported URL formats:
  GitHub:
    https://github.com/{owner}/{repo}
    https://github.com/{owner}/{repo}/tree/{branch}
  GitLab (wikimedia or gitlab.com):
    https://gitlab.wikimedia.org/{namespace}/{repo}
    https://gitlab.com/{namespace}/{repo}
"""

import re
import base64
import requests

ALLOWED_EXTENSIONS = {".js", ".css"}
DEFAULT_TIMEOUT = 10  # seconds


class RepoFetchError(Exception):
    """Raised when a repository cannot be fetched or parsed."""


# ──────────────────────────────────────────────────────────────────────────────
# Public entry point
# ──────────────────────────────────────────────────────────────────────────────

def fetch_files(repo_url: str) -> list[dict]:
    """
    Given a public repository URL, return a list of JS/CSS file dicts.

    Each dict contains:
        name        str   relative path inside the repo  e.g. "src/gadget.js"
        content     str   raw UTF-8 file content
        size        int   file size in bytes
        platform    str   "github" | "gitlab"
    """
    platform, owner, repo, branch = _parse_url(repo_url)

    if platform == "github":
        return _fetch_github(owner, repo, branch)
    elif platform == "gitlab":
        base = _gitlab_base(repo_url)
        return _fetch_gitlab(base, owner, repo, branch)
    else:
        raise RepoFetchError(f"Unsupported repository URL: {repo_url}")


# ──────────────────────────────────────────────────────────────────────────────
# URL parsing
# ──────────────────────────────────────────────────────────────────────────────

def _parse_url(url: str):
    """
    Returns (platform, owner, repo, branch).
    branch is None if not specified in the URL (callers should default to HEAD).
    """
    url = url.rstrip("/")

    # GitHub
    gh = re.match(
        r"https://github\.com/([^/]+)/([^/]+?)(?:\.git)?(?:/tree/([^/]+))?$",
        url,
    )
    if gh:
        return "github", gh.group(1), gh.group(2), gh.group(3)

    # GitLab (wikimedia or gitlab.com)
    gl = re.match(
        r"https://(gitlab\.wikimedia\.org|gitlab\.com)/([^/]+(?:/[^/]+)*)/([^/]+?)(?:\.git)?(?:/-/tree/([^/]+))?$",
        url,
    )
    if gl:
        # gl.group(1) = host, gl.group(2) = namespace, gl.group(3) = repo, gl.group(4) = branch
        namespace = gl.group(2)
        repo = gl.group(3)
        branch = gl.group(4)
        host = gl.group(1)
        return "gitlab", namespace, repo, branch

    raise RepoFetchError(
        "Could not parse repository URL. "
        "Expected a public GitHub or GitLab URL."
    )


def _gitlab_base(url: str) -> str:
    """Extract the GitLab instance base URL (e.g. https://gitlab.wikimedia.org)."""
    m = re.match(r"(https://[^/]+)", url)
    if m:
        return m.group(1)
    return "https://gitlab.com"


# ──────────────────────────────────────────────────────────────────────────────
# GitHub fetcher
# ──────────────────────────────────────────────────────────────────────────────

def _fetch_github(owner: str, repo: str, branch: str | None) -> list[dict]:
    """
    Uses the GitHub Trees API to get the full recursive file list,
    then fetches content only for JS/CSS files.

    API docs: https://docs.github.com/en/rest/git/trees
    Rate limit: 60 unauthenticated requests/hour per IP.
    """
    # Resolve branch if not specified
    if not branch:
        branch = _github_default_branch(owner, repo)

    tree_url = f"https://api.github.com/repos/{owner}/{repo}/git/trees/{branch}?recursive=1"
    response = _get(tree_url, platform="GitHub")
    data = response.json()

    if data.get("truncated"):
        # Repository has more than 100,000 files — fetch top-level only
        pass  # the filter below will still work on what we have

    files = []
    for item in data.get("tree", []):
        if item.get("type") != "blob":
            continue
        path = item["path"]
        if not _is_allowed(path):
            continue

        content = _github_file_content(owner, repo, branch, path)
        if content is not None:
            files.append({
                "name": path,
                "content": content,
                "size": item.get("size", 0),
                "platform": "github",
            })

    return files


def _github_default_branch(owner: str, repo: str) -> str:
    url = f"https://api.github.com/repos/{owner}/{repo}"
    response = _get(url, platform="GitHub")
    return response.json().get("default_branch", "main")


def _github_file_content(owner: str, repo: str, branch: str, path: str) -> str | None:
    """Fetch a single file's content from GitHub using the Contents API."""
    url = f"https://api.github.com/repos/{owner}/{repo}/contents/{path}?ref={branch}"
    try:
        response = _get(url, platform="GitHub")
        data = response.json()
        encoded = data.get("content", "")
        # GitHub returns base64-encoded content with newlines
        decoded = base64.b64decode(encoded.replace("\n", "")).decode("utf-8", errors="replace")
        return decoded
    except Exception:
        return None


# ──────────────────────────────────────────────────────────────────────────────
# GitLab fetcher
# ──────────────────────────────────────────────────────────────────────────────

def _fetch_gitlab(base: str, namespace: str, repo: str, branch: str | None) -> list[dict]:
    """
    Uses the GitLab Repository Tree and Raw File APIs.

    API docs: https://docs.gitlab.com/ee/api/repositories.html
    Works for both gitlab.com and gitlab.wikimedia.org (unauthenticated, public repos).
    """
    # URL-encode the full project path (namespace/repo)
    from urllib.parse import quote
    project_path = quote(f"{namespace}/{repo}", safe="")

    # Resolve default branch
    if not branch:
        branch = _gitlab_default_branch(base, project_path)

    # Fetch full file tree recursively
    tree_url = (
        f"{base}/api/v4/projects/{project_path}/repository/tree"
        f"?recursive=true&per_page=100&ref={branch}"
    )

    all_items = []
    page = 1
    while True:
        response = _get(f"{tree_url}&page={page}", platform="GitLab")
        items = response.json()
        if not items:
            break
        all_items.extend(items)
        # GitLab paginates — check if there are more pages
        if len(items) < 100:
            break
        page += 1

    files = []
    for item in all_items:
        if item.get("type") != "blob":
            continue
        path = item["path"]
        if not _is_allowed(path):
            continue

        content = _gitlab_file_content(base, project_path, branch, path)
        if content is not None:
            files.append({
                "name": path,
                "content": content,
                "size": len(content.encode("utf-8")),
                "platform": "gitlab",
            })

    return files


def _gitlab_default_branch(base: str, project_path: str) -> str:
    from urllib.parse import quote
    url = f"{base}/api/v4/projects/{project_path}"
    response = _get(url, platform="GitLab")
    return response.json().get("default_branch", "main")


def _gitlab_file_content(base: str, project_path: str, branch: str, path: str) -> str | None:
    """Fetch a single file's raw content from GitLab."""
    from urllib.parse import quote
    encoded_path = quote(path, safe="")
    url = f"{base}/api/v4/projects/{project_path}/repository/files/{encoded_path}/raw?ref={branch}"
    try:
        response = _get(url, platform="GitLab")
        return response.text
    except Exception:
        return None

def _is_allowed(path: str) -> bool:
    """Return True only for .js and .css files."""
    return any(path.endswith(ext) for ext in ALLOWED_EXTENSIONS)


def _get(url: str, platform: str = "remote") -> requests.Response:
    """
    Perform a GET request with a descriptive User-Agent and error handling.
    Raises RepoFetchError on non-2xx responses.
    """
    headers = {
        "User-Agent": "WikiScriptPublisher/1.0 (https://script-publisher.toolforge.org; Toolforge)",
        "Accept": "application/json",
    }
    try:
        response = requests.get(url, headers=headers, timeout=DEFAULT_TIMEOUT)
    except requests.exceptions.ConnectionError:
        raise RepoFetchError(f"Could not connect to {platform}. Check your network.")
    except requests.exceptions.Timeout:
        raise RepoFetchError(f"{platform} request timed out after {DEFAULT_TIMEOUT}s.")

    if response.status_code == 404:
        raise RepoFetchError(
            f"Repository not found on {platform} (404). "
            "Make sure the repository is public and the URL is correct."
        )
    if response.status_code == 403:
        raise RepoFetchError(
            f"{platform} rate limit reached (403). "
            "Please wait a few minutes and try again."
        )
    if not response.ok:
        raise RepoFetchError(
            f"{platform} returned an unexpected error: "
            f"HTTP {response.status_code}"
        )

    return response

      - BotPassword      (credential stored in session, used at publish time)

    BotPassword entered here is stored in the session only.

        request.session["user_username"] = wiki_username
        request.session["user_groups"] = []
        request.session["user_rights"] = []
        request.session["user_token"] = None
        request.session["bot_username"] = bot_username
        request.session["bot_password"] = bot_password  # session is server-side only
        request.session.set_expiry(86400)  # 24 hours
        return redirect("dashboard")

    error = request.session.pop("auth_error", None)
    return render(request, "publisher/get_started.html", {"error": error})

def oauth_callback(request):
    """
    Step 2: Wikimedia redirects here after the user authorizes (or denies).

    On success: stores user info in session, redirects to dashboard.

        return redirect("dashboard")
    except OAuthError as e:

        return redirect("get_started")

@csrf_exempt
@require_POST
def api_fetch_repo(request):
    """
    POST /api/fetch-repo/
    Body: {"repo_url": "https://github.com/owner/repo"}

    Returns:
        {"files": [{"name": "...", "content": "...", "size": 123, "platform": "github"}]}
    or
        {"error": "..."}  with an appropriate HTTP status code.

    Fetches files directly from GitHub/GitLab raw APIs.
    No third-party services are called.
    """
    if not get_current_user(request):
        return JsonResponse({"error": "Not authenticated."}, status=401)

    try:
        body = json.loads(request.body)
    except json.JSONDecodeError:
        return JsonResponse({"error": "Invalid JSON body."}, status=400)

    repo_url = body.get("repo_url", "").strip()
    if not repo_url:
        return JsonResponse({"error": "repo_url is required."}, status=400)

    try:
        files = fetch_files(repo_url)
    except RepoFetchError as e:
        return JsonResponse({"error": str(e)}, status=422)
    except Exception as e:
        # Log unexpected errors but don't expose internals
        import logging
        logging.getLogger(__name__).exception("Unexpected error in api_fetch_repo")
        return JsonResponse({"error": "An unexpected error occurred while fetching the repository."}, status=500)

    return JsonResponse({"files": files, "count": len(files)})

# API: Publish (Phase 3 — BotPassword path, and Phase 4 — draft path)
# ──────────────────────────────────────────────────────────────────────────────

@csrf_exempt
@require_POST
def api_publish(request):
    """
    POST /api/publish/
    Body:
    {
        "method":         "botpassword" | "draft",
        "bot_username":   "Username@BotName",   // only for botpassword
        "bot_password":   "the-password",       // only for botpassword
        "wiki":           "en.wikipedia.org",
        "page":           "User:Dev/script.js",
        "content":        "... full file content ...",
        "edit_summary":   "Update from Script Publisher v1.0",
        "source_repo":    "https://github.com/owner/repo",
        "source_file":    "src/script.js",
        "version":        "v1.0.0"
    }

    BotPassword path:
        Logs in with the provided credentials and edits the page directly.
        The BotPassword is used only for this request and never stored.

    Draft path:
        Does not edit anything.
        Returns a pre-filled edit URL for the user to open in their browser.
    """
    current_user = get_current_user(request)
    if not current_user:
        return JsonResponse({"error": "Not authenticated."}, status=401)

    try:
        body = json.loads(request.body)
    except json.JSONDecodeError:
        return JsonResponse({"error": "Invalid JSON body."}, status=400)

    method = body.get("method", "")
    wiki = body.get("wiki", "").strip()
    page = body.get("page", "").strip()
    content = body.get("content", "")
    edit_summary = body.get("edit_summary", "Published via Script Publisher").strip()
    source_repo = body.get("source_repo", "").strip()
    source_file = body.get("source_file", "").strip()
    version = body.get("version", "").strip()

    if not wiki or not page or not content:
        return JsonResponse(
            {"error": "wiki, page, and content are required."}, status=400
        )

    if method == "botpassword":
        return _publish_via_botpassword(
            request, current_user, body,
            wiki, page, content, edit_summary,
            source_repo, source_file, version,
        )
    elif method == "draft":
        return _publish_via_draft(
            current_user, wiki, page, content, edit_summary,
            source_repo, source_file, version,
        )
    else:
        return JsonResponse(
            {"error": "method must be 'botpassword' or 'draft'."}, status=400
        )


def _publish_via_botpassword(
    request, current_user, body,
    wiki, page, content, edit_summary,
    source_repo, source_file, version,
):
    """
    Publish using the user's own BotPassword credentials.

    Security notes:
    - The BotPassword is used only for this HTTP request.
    - It is never written to the database or logged.
    - The user must have confirmed the diff before this endpoint is called.
    - The tool only performs the edit the user explicitly requested.
    """
    bot_username = body.get("bot_username", "").strip()
    bot_password = body.get("bot_password", "")

    if not bot_username or not bot_password:
        return JsonResponse(
            {"error": "bot_username and bot_password are required for BotPassword publish."},
            status=400,
        )

    log_entry = PublishLog(
        username=current_user["username"],
        source_repo=source_repo,
        source_file=source_file,
        target_wiki=wiki,
        target_page=page,
        version=version,
        edit_summary=edit_summary,
        publish_method="botpassword",
        status="pending_review",
    )

    try:
        session = BotSession(wiki)
        session.login(bot_username, bot_password)
        result = session.edit_page(page, content, edit_summary)

        log_entry.status = "success"
        log_entry.save()

        return JsonResponse({
            "success": True,
            "method": "botpassword",
            "wiki": wiki,
            "page": page,
            "revision_id": result.get("newrevid"),
            "log_id": log_entry.id,
        })

    except WikiAPIError as e:
        log_entry.status = "failed"
        log_entry.error_message = str(e)
        log_entry.save()
        return JsonResponse({"error": str(e)}, status=422)

    except Exception:
        log_entry.status = "failed"
        log_entry.error_message = "Unexpected internal error."
        log_entry.save()
        import logging
        logging.getLogger(__name__).exception("Unexpected error in BotPassword publish")
        return JsonResponse(
            {"error": "An unexpected error occurred during publishing."}, status=500
        )


def _publish_via_draft(
    current_user, wiki, page, content, edit_summary,
    source_repo, source_file, version,
):
    """
    Generate a draft edit URL instead of editing the page directly.

    This path is safe for users without interface-admin rights,
    and requires no special OAuth permissions.
    """
    edit_url = build_draft_edit_url(wiki, page, content)
    notification = build_talk_page_notification(
        wiki, page, source_repo,
        current_user["username"], edit_summary,
    )

    # Log the draft action
    PublishLog.objects.create(
        username=current_user["username"],
        source_repo=source_repo,
        source_file=source_file,
        target_wiki=wiki,
        target_page=page,
        version=version,
        edit_summary=edit_summary,
        publish_method="draft",
        status="pending_review",
    )

    return JsonResponse({
        "success": True,
        "method": "draft",
        "edit_url": edit_url,
        "notification_wikitext": notification,
        "instructions": (
            "Open the edit URL in your browser to review and submit the change. "
            "Copy the notification wikitext to the relevant talk page."
        ),
    })


==== publisher\wiki_api.py ====
"""
wiki_api.py
-----------
Handles all MediaWiki API interactions for Script Publisher.

Two publish paths are provided:

  Path A — BotPassword
    The user supplies their own BotPassword credentials.
    The tool logs in as that bot and publishes on the user's behalf.
    This requires no special community approval — it is equivalent
    to the user running a script locally.

  Path B — Notification + draft (safe MVP path)
    Instead of editing the wiki page directly, the tool generates:
      1. A pre-filled edit URL the user can open in their browser.
      2. An optional talk-page notification with the proposed change.
    The user reviews and submits the edit themselves.
    This path requires zero OAuth JS-editing permissions.

Both paths require explicit user confirmation before any edit is made.
No automated or scheduled publishing is performed.
"""

import re
import requests
from urllib.parse import urljoin

REQUEST_TIMEOUT = 15  # seconds

# Wikis that the tool supports — expand as needed
KNOWN_WIKIS = {
    "commons.wikimedia.org": "https://commons.wikimedia.org/w/api.php",
    "en.wikipedia.org": "https://en.wikipedia.org/w/api.php",
    "de.wikipedia.org": "https://de.wikipedia.org/w/api.php",
    "fr.wikipedia.org": "https://fr.wikipedia.org/w/api.php",
    "es.wikipedia.org": "https://es.wikipedia.org/w/api.php",

    "meta.wikimedia.org": "https://meta.wikimedia.org/w/api.php",
}


class WikiAPIError(Exception):
    """Raised when a MediaWiki API call fails."""


# ──────────────────────────────────────────────────────────────────────────────
# BotPassword authentication
# ──────────────────────────────────────────────────────────────────────────────

class BotSession:
    """
    A persistent session authenticated with a BotPassword.

    Usage:
        session = BotSession("en.wikipedia.org")
        session.login("MyBot@my-bot-password", "the-password-string")
        session.edit_page("User:MyBot/script.js", new_content, "Update from Script Publisher")
    """

    def __init__(self, wiki_domain: str):
        self.api_url = _api_url(wiki_domain)
        self.session = requests.Session()
        self.session.headers.update({"User-Agent": _user_agent()})
        self._logged_in = False

    def login(self, bot_username: str, bot_password: str):
        """
        Authenticate using a BotPassword.

        BotPassword format:   "Username@BotName"
        Actual password:      the generated password from Special:BotPasswords

        Two-step login as required by the MediaWiki API:
          1. Get login token
          2. POST credentials with token
        """
        # Step 1: get login token
        token = self._get_token("login")

        # Step 2: submit credentials
        response = self._post({
            "action": "login",
            "lgname": bot_username,
            "lgpassword": bot_password,
            "lgtoken": token,
            "format": "json",
        })

        result = response.json()
        login_result = result.get("login", {}).get("result")

        if login_result == "Success":
            self._logged_in = True
        elif login_result == "Failed":
            reason = result.get("login", {}).get("reason", "unknown reason")
            raise WikiAPIError(f"BotPassword login failed: {reason}")
        else:
            raise WikiAPIError(
                f"Unexpected login result: {login_result}. "
                f"Full response: {result}"
            )

    def get_page_content(self, page_title: str) -> str | None:
        """
        Fetch the current wikitext content of a page.
        Returns None if the page does not exist.
        """
        response = self._get({
            "action": "query",
            "prop": "revisions",
            "titles": page_title,
            "rvprop": "content",
            "rvslots": "main",
            "format": "json",
        })
        data = response.json()
        pages = data.get("query", {}).get("pages", {})

        for page_id, page_data in pages.items():
            if page_id == "-1":
                return None  # Page does not exist
            revisions = page_data.get("revisions", [])
            if revisions:
                return revisions[0].get("slots", {}).get("main", {}).get("*", "")
        return None

    def edit_page(
        self,
        page_title: str,
        new_content: str,
        edit_summary: str,
        minor: bool = False,
    ) -> dict:
        """
        Replace the content of a wiki page.

        Returns the API response dict on success.
        Raises WikiAPIError on failure.

        Security contract:
          - This method is only called after the user has confirmed the diff.
          - The caller is responsible for ensuring the user has the rights
            to edit the target page.
        """
        if not self._logged_in:
            raise WikiAPIError("Not logged in. Call login() first.")

        # Get a CSRF token
        csrf_token = self._get_token("csrf")

        response = self._post({
            "action": "edit",
            "title": page_title,
            "text": new_content,
            "summary": edit_summary,
            "minor": "1" if minor else "0",
            "bot": "1",
            "token": csrf_token,
            "format": "json",
        })

        data = response.json()

        if "error" in data:
            code = data["error"].get("code", "unknown")
            info = data["error"].get("info", "")
            raise WikiAPIError(f"Edit failed [{code}]: {info}")

        edit_result = data.get("edit", {})
        if edit_result.get("result") != "Success":
            raise WikiAPIError(f"Edit did not succeed: {data}")

        return edit_result

    # ── Internal helpers ─────────────────────────────────────────────────────

    def _get_token(self, token_type: str) -> str:
        response = self._get({
            "action": "query",
            "meta": "tokens",
            "type": token_type,
            "format": "json",
        })
        data = response.json()
        token = data.get("query", {}).get("tokens", {}).get(f"{token_type}token")
        if not token:
            raise WikiAPIError(f"Could not retrieve {token_type} token.")
        return token

    def _get(self, params: dict) -> requests.Response:
        try:
            response = self.session.get(
                self.api_url, params=params, timeout=REQUEST_TIMEOUT
            )
        except requests.exceptions.RequestException as e:
            raise WikiAPIError(f"Network error: {e}")
        if not response.ok:
            raise WikiAPIError(f"HTTP {response.status_code} from MediaWiki API.")
        return response

    def _post(self, data: dict) -> requests.Response:
        try:
            response = self.session.post(
                self.api_url, data=data, timeout=REQUEST_TIMEOUT
            )
        except requests.exceptions.RequestException as e:
            raise WikiAPIError(f"Network error: {e}")
        if not response.ok:
            raise WikiAPIError(f"HTTP {response.status_code} from MediaWiki API.")
        return response

# Path B — Notification + draft URL (no direct JS-editing required)
# ──────────────────────────────────────────────────────────────────────────────

def build_draft_edit_url(wiki_domain: str, page_title: str, content: str) -> str:
    """
    Build a pre-filled edit URL for the user to open in their browser.

    The user sees the content pre-loaded in the edit form, reviews it,
    and submits it themselves. Script Publisher never touches the page.

    This is the safe MVP publish path — no OAuth JS-editing needed.
    """
    from urllib.parse import urlencode, quote

    base = f"https://{wiki_domain}/wiki/{quote(page_title, safe='')}"
    params = {
        "action": "edit",
        "preloadtitle": page_title,
    }
    # Note: preload with full content is too long for a URL parameter.
    # We provide the edit URL + instructions to paste the content.
    # A future version can use the API's createonly=true flow.
    return f"https://{wiki_domain}/w/index.php?title={quote(page_title, safe='')}&action=edit"


def build_talk_page_notification(
    wiki_domain: str,
    page_title: str,
    repo_url: str,
    username: str,
    edit_summary: str,
) -> str:
    """
    Build a wikitext notification message for the relevant talk page.

    This is posted to inform community members that a new version
    is available from the source repository.
    """
    return (
        f"== Script update available ==\n\n"
        f"A new version of [[{page_title}]] is available from "
        f"[{repo_url} the source repository].\n\n"
        f"; Edit summary: {edit_summary}\n"
        f"; Prepared by: [[User:{username}]] using "
        f"[[Toolforge:script-publisher|Script Publisher]]\n\n"
        f"Please review the diff before applying. ~~~~"
    )


# ──────────────────────────────────────────────────────────────────────────────
# Diff helper
# ──────────────────────────────────────────────────────────────────────────────

def compute_diff(old_content: str | None, new_content: str) -> list[dict]:
    """
    Compute a simple line-by-line diff between old and new content.

    Returns a list of dicts:
        {"type": "added"|"removed"|"unchanged", "line": str}

    Used to show the user what will change before they confirm publish.
    This is a naive diff — for production consider using difflib.unified_diff.
    """
    import difflib

    old_lines = (old_content or "").splitlines()
    new_lines = new_content.splitlines()

    diff = []
    matcher = difflib.SequenceMatcher(None, old_lines, new_lines)

    for tag, i1, i2, j1, j2 in matcher.get_opcodes():
        if tag == "equal":
            for line in old_lines[i1:i2]:
                diff.append({"type": "unchanged", "line": line})
        elif tag in ("replace", "delete"):
            for line in old_lines[i1:i2]:
                diff.append({"type": "removed", "line": line})
            if tag == "replace":
                for line in new_lines[j1:j2]:
                    diff.append({"type": "added", "line": line})
        elif tag == "insert":
            for line in new_lines[j1:j2]:
                diff.append({"type": "added", "line": line})

    return diff

def _api_url(wiki_domain: str) -> str:
    """Return the MediaWiki API endpoint for a given domain."""
    clean = wiki_domain.lower().strip()
    if clean in KNOWN_WIKIS:
        return KNOWN_WIKIS[clean]
    # Fallback: construct the URL from the domain
    return f"https://{clean}/w/api.php"


def _user_agent() -> str:
    return "WikiScriptPublisher/1.0 (https://script-publisher.toolforge.org; Toolforge)"

  const WIKI_CHOICES = {{ wiki_choices|safe }};

    const fresh = sessionStorage.getItem("selectedFiles");

    if (saved) {

        `<option value="${w}" ${file.wiki===w?"selected":""}>${w}</option>`
      ).join("");

      const tr = document.createElement("tr");
      tr.innerHTML = `
        <td class="sp-mono sp-text-small">${escHtml(file.source)}</td>
        <td>
          <select class="sp-select" style="font-size:.8125rem;padding:.3rem .5rem;"

            <i class="fa-regular fa-trash-can"></i>
          </button>
        </td>
      `;
      tbody.appendChild(tr);
    });
  }

  function removeFile(i) {
    const name = fileMappings[i].source;
    fileMappings.splice(i, 1);
    renderTable();
    toast(`Removed ${name}`);
  }

    let count = 0;
    fileMappings.forEach(f => {
      if (!f.dest) {
        f.dest = "MediaWiki:Gadget-" + f.source.split("/").pop();
        count++;
      }
    });
    renderTable();
    toast(`Auto-suggested ${count} path(s)`);
  }

    fileMappings.forEach(f => {

  }

    document.getElementById("prefixModal").style.display = "none";

    return String(s||"").replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;").replace(/"/g,"&quot;");

<script>
  const CSRF_TOKEN = "{{ csrf_token }}";
  const WIKI_USERNAME = "{{ current_user.username }}";
  let mappings = [];      // [{source, wiki, dest, content}]
  let currentMethod = "draft";

  // ── Load mappings from session ────────────────────────────────────────────

  document.addEventListener("DOMContentLoaded", () => {
    const raw = sessionStorage.getItem("mappingConfig");
    const files = sessionStorage.getItem("selectedFiles");

    if (raw) {
      const savedMappings = JSON.parse(raw);
      const fileMap = {};
      if (files) {
        JSON.parse(files).forEach(f => { fileMap[f.name] = f.content; });
      }
      mappings = savedMappings.map(m => ({
        ...m,
        content: fileMap[m.source] || "",
      }));
      renderFileSummary();
    }
  });

  function renderFileSummary() {
    const container = document.getElementById("fileSummaryList");
    if (!mappings.length) return;
    container.innerHTML = "";
    mappings.forEach(m => {
      const div = document.createElement("div");
      div.className = "text-sm border-b border-gray-100 pb-3 last:border-0 last:pb-0";
      div.innerHTML = `
        <p class="font-medium text-gray-900 font-mono truncate">${escapeHtml(m.source)}</p>
        <p class="text-xs text-gray-500 mt-0.5">
          → <span class="font-mono">${escapeHtml(m.dest || "(no destination set)")}</span>
        </p>
        <p class="text-xs text-gray-400">${escapeHtml(m.wiki)}</p>
      `;
      container.appendChild(div);
    });
  }

  // ── Method toggle ─────────────────────────────────────────────────────────

  function switchMethod(method) {
    currentMethod = method;
    const activeClasses = ["bg-white","text-gray-900","shadow-sm","font-semibold"];
    const inactiveClasses = ["text-gray-500","hover:text-gray-900","font-medium"];

    ["draft","botpassword"].forEach(m => {
      const btn = document.getElementById(`tab-${m}`);
      const panel = document.getElementById(`method-${m}`);
      if (m === method) {
        btn.classList.add(...activeClasses);
        btn.classList.remove(...inactiveClasses);
        panel.classList.remove("hidden");
      } else {
        btn.classList.remove(...activeClasses);
        btn.classList.add(...inactiveClasses);
        panel.classList.add("hidden");
      }
    });
  }

  // ── Start publish — fetch diff, show confirmation modal ───────────────────

  async function startPublish() {
    const summary = document.getElementById("metaSummary").value.trim();
    if (!summary) {
      showPublishError("Edit summary is required.");
      return;
    }
    if (!mappings.length) {
      showPublishError("No files to publish. Go back to mapping config.");
      return;
    }
    const unmapped = mappings.filter(m => !m.dest || !m.wiki);
    if (unmapped.length) {
      showPublishError(`${unmapped.length} file(s) have no destination set. Complete the mapping config first.`);
      return;
    }

    hidePublishError();

    // Build diff display for all files
    const diffHtml = await buildDiffHtml();
    document.getElementById("diffContainer").innerHTML = diffHtml;
    document.getElementById("confirmModal").classList.remove("hidden");
  }

  async function buildDiffHtml() {
    let html = "";
    for (const m of mappings) {
      // Fetch current wiki content for diff
      let currentContent = null;
      try {
        const r = await fetch(`/api/page-content/?wiki=${encodeURIComponent(m.wiki)}&page=${encodeURIComponent(m.dest)}`);
        if (r.ok) {
          const d = await r.json();
          currentContent = d.content;
        }
      } catch (_) {}

      const newLines = (m.content || "").split("\n");
      const oldLines = (currentContent || "").split("\n");

      html += `
        <div class="border-b border-gray-200 last:border-0">
          <div class="px-6 py-3 bg-gray-50 border-b border-gray-200 flex justify-between items-center">
            <span class="text-sm font-bold font-mono text-gray-900">${escapeHtml(m.source)}</span>
            <span class="text-xs text-gray-500">→ ${escapeHtml(m.dest)} on ${escapeHtml(m.wiki)}</span>
          </div>
          <div class="flex">
            <div class="flex-1 min-w-0 border-r border-gray-200">
              <div class="px-4 py-2 bg-red-50 text-xs font-bold text-red-800 border-b border-red-100">
                Current on wiki ${currentContent === null ? "(page does not exist yet)" : ""}
              </div>
              <pre class="px-4 py-3 text-xs font-mono overflow-auto bg-white text-gray-600 max-h-48 whitespace-pre-wrap">${escapeHtml(currentContent || "")}</pre>
            </div>
            <div class="flex-1 min-w-0">
              <div class="px-4 py-2 bg-green-50 text-xs font-bold text-green-800 border-b border-green-100">
                Incoming from repository
              </div>
              <pre class="px-4 py-3 text-xs font-mono overflow-auto bg-white text-gray-800 max-h-48 whitespace-pre-wrap">${escapeHtml(m.content || "")}</pre>
            </div>
          </div>
        </div>
      `;
    }
    return html;
  }

  function closeConfirmModal() {
    document.getElementById("confirmModal").classList.add("hidden");
  }

  // ── Execute publish after confirmation ────────────────────────────────────

  async function confirmAndPublish() {
    const confirmBtn = document.getElementById("confirmBtn");
    confirmBtn.disabled = true;
    confirmBtn.innerHTML = '<i class="fa-solid fa-circle-notch fa-spin"></i> Publishing…';

    const summary   = document.getElementById("metaSummary").value.trim();
    const version   = document.getElementById("metaVersion").value.trim();
    const sourceRepo = JSON.parse(sessionStorage.getItem("selectedFiles") || "[]")[0]?.platform || "";

    const results = [];
    for (const m of mappings) {
      const payload = {
        method: currentMethod,
        wiki: m.wiki,
        page: m.dest,
        content: m.content,
        edit_summary: summary,
        source_repo: sessionStorage.getItem("repoUrl") || "",
        source_file: m.source,
        version: version,
      };

      if (currentMethod === "botpassword") {
        payload.bot_username = document.getElementById("botUsername")?.value?.trim() || "";
        payload.bot_password = document.getElementById("botPassword")?.value || "";
      }

      try {
        const r = await fetch("/api/publish/", {
          method: "POST",
          headers: {
            "Content-Type": "application/json",
            "X-CSRFToken": CSRF_TOKEN,
          },
          body: JSON.stringify(payload),
        });
        const d = await r.json();
        results.push({ file: m.source, ok: r.ok, data: d });
      } catch (e) {
        results.push({ file: m.source, ok: false, data: { error: "Network error." } });
      }
    }

    closeConfirmModal();

    const firstResult = results[0]?.data;
    if (currentMethod === "draft" && firstResult?.edit_url) {
      document.getElementById("draftEditUrl").value = firstResult.edit_url;
      document.getElementById("draftEditLink").href = firstResult.edit_url;
      document.getElementById("draftNotification").value = firstResult.notification_wikitext || "";
      document.getElementById("draftResultModal").classList.remove("hidden");
    } else {
      const failures = results.filter(r => !r.ok);
      if (failures.length === 0) {
        alert(`Published ${results.length} file(s) successfully.`);
        window.location.href = "{% url 'publish_log' %}";
      } else {
        showPublishError(
          `${failures.length} file(s) failed: ` +
          failures.map(f => `${f.file}: ${f.data.error}`).join("; ")
        );
      }
    }

    confirmBtn.disabled = false;
    confirmBtn.innerHTML = '<i class="fa-solid fa-check"></i> Confirm and publish';
  }

  function copyNotification() {
    const text = document.getElementById("draftNotification").value;
    navigator.clipboard.writeText(text).then(() => {
      alert("Notification wikitext copied to clipboard.");
    });
  }

  function showPublishError(msg) {
    document.getElementById("publishError").classList.remove("hidden");
    document.getElementById("publishErrorText").textContent = msg;
  }

  function hidePublishError() {
    document.getElementById("publishError").classList.add("hidden");
  }

  function escapeHtml(str) {
    if (!str) return "";
    return String(str).replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;");
  }
</script>
{% endblock %}

<script>
  const CSRF_TOKEN = "{{ csrf_token }}";

  // Files fetched from the API — kept in memory until the user navigates away
  let fetchedFiles = [];

  async function fetchRepo() {
    const repoUrl = document.getElementById("repoInput").value.trim();
    if (!repoUrl) {
      showError("Please enter a repository URL.");
      return;
    }

    setLoading(true);
    hideError();
    document.getElementById("fileSection").classList.add("hidden");
    document.getElementById("fileTableBody").innerHTML = "";
    fetchedFiles = [];

    try {
      const response = await fetch("/api/fetch-repo/", {
        method: "POST",
        headers: {
          "Content-Type": "application/json",
          "X-CSRFToken": CSRF_TOKEN,
        },
        body: JSON.stringify({ repo_url: repoUrl }),
      });

      const data = await response.json();

      if (!response.ok) {
        showError(data.error || "Failed to fetch repository.");
        return;
      }

      if (!data.files || data.files.length === 0) {
        showError("No .js or .css files found in this repository.");
        return;
      }

      fetchedFiles = data.files;
      renderFileTable(data.files);
      document.getElementById("fileSection").classList.remove("hidden");
      document.getElementById("fileCount").textContent =
        `${data.files.length} file${data.files.length === 1 ? "" : "s"} found`;

    } catch (err) {
      showError("Could not connect to the server. Please try again.");
      console.error(err);
    } finally {
      setLoading(false);
    }
  }

  function renderFileTable(files) {
    const tbody = document.getElementById("fileTableBody");
    tbody.innerHTML = "";

    files.forEach((file, index) => {
      const isJs = file.name.endsWith(".js");
      const iconClass = isJs ? "fa-square-js text-yellow-500" : "fa-css3 text-blue-500";
      const sizeKb = (file.size / 1024).toFixed(1);

      const tr = document.createElement("tr");
      tr.className = "hover:bg-gray-50 transition-colors";
      tr.innerHTML = `
        <td class="px-4 py-3">
          <input type="checkbox" checked data-index="${index}"
            class="file-checkbox h-4 w-4 text-brand-600 focus:ring-brand-500 border-gray-300 rounded"
          />
        </td>
        <td class="px-4 py-3">
          <div class="flex items-center gap-2">
            <i class="fa-brands ${iconClass} text-lg flex-shrink-0"></i>
            <span class="text-sm font-medium text-gray-900 font-mono">${escapeHtml(file.name)}</span>
            <span class="text-xs text-gray-400">${escapeHtml(file.platform)}</span>
          </div>
        </td>
        <td class="px-4 py-3 text-center text-xs text-gray-500">${sizeKb} KB</td>
        <td class="px-4 py-3 text-center">
          <button
            onclick="openModal(${index})"
            class="text-gray-400 hover:text-brand-600 transition-colors"
            title="Preview file content"
          >
            <i class="fa-regular fa-eye"></i>
          </button>
        </td>
      `;
      tbody.appendChild(tr);
    });
  }

  function saveAndContinue() {
    const checkboxes = document.querySelectorAll(".file-checkbox:checked");
    const selected = Array.from(checkboxes).map(cb => {
      const idx = parseInt(cb.dataset.index, 10);
      return {
        name: fetchedFiles[idx].name,
        content: fetchedFiles[idx].content,
        size: fetchedFiles[idx].size,
        platform: fetchedFiles[idx].platform,
      };
    });

    if (selected.length === 0) {
      alert("Please select at least one file to continue.");
      return;
    }

    sessionStorage.setItem("selectedFiles", JSON.stringify(selected));
    window.location.href = "{% url 'mapping_config' %}";
  }

  function openModal(index) {
    const file = fetchedFiles[index];
    document.getElementById("modalTitle").textContent = file.name;
    document.getElementById("modalContent").textContent = file.content;
    document.getElementById("previewModal").classList.remove("hidden");
  }

  function closeModal() {
    document.getElementById("previewModal").classList.add("hidden");
  }

  function toggleAll(source) {
    document.querySelectorAll(".file-checkbox").forEach(cb => {
      cb.checked = source.checked;
    });
  }

  function setLoading(on) {
    document.getElementById("btnText").textContent = on ? "Fetching…" : "Fetch files";
    document.getElementById("btnLoader").classList.toggle("hidden", !on);
    document.getElementById("fetchBtn").disabled = on;
  }

  function showError(msg) {
    document.getElementById("fetchError").classList.remove("hidden");
    document.getElementById("fetchErrorText").textContent = msg;
  }

  function hideError() {
    document.getElementById("fetchError").classList.add("hidden");
  }

  function escapeHtml(str) {
    return str.replace(/&/g,"&amp;").replace(/</g,"&lt;").replace(/>/g,"&gt;");
  }
</script>

"""
notifier.py
-----------
Posts change notifications to Wikimedia user talk pages.

Based on the approach discussed with KC (mentor):
  - The tool tracks which scripts are in use and where
  - When the source repository changes, a bot posts a message to the
    affected user's talk page on the target wiki
  - Users opt in to receive these notifications
  - Users can opt out via a link in the notification message

The bot that posts these messages must be a DEDICATED Wikimedia account
(not the developer's personal account). Create a new account like
"WikiScriptSyncBot" and generate a BotPassword for it at
Special:BotPasswords on the target wiki.

Set these in your .env:
  NOTIFIER_BOT_USERNAME=WikiScriptSyncBot@notifier-app
  NOTIFIER_BOT_PASSWORD=the-generated-password
  NOTIFIER_BOT_WIKI=meta.wikimedia.org
"""

import hashlib
import logging
import os
from datetime import datetime, timezone

import requests
from django.conf import settings
from django.utils import timezone as django_timezone

from .wiki_api import BotSession, WikiAPIError

logger = logging.getLogger(__name__)

BOT_USERNAME = os.environ.get("NOTIFIER_BOT_USERNAME", "")
BOT_PASSWORD = os.environ.get("NOTIFIER_BOT_PASSWORD", "")
BOT_WIKI     = os.environ.get("NOTIFIER_BOT_WIKI", "meta.wikimedia.org")

TOOLFORGE_URL = "https://script-publisher.toolforge.org"


# ──────────────────────────────────────────────────────────────────────────────
# Public functions
# ──────────────────────────────────────────────────────────────────────────────

def check_and_notify_all():
    """
    Check every active TrackedScript for upstream changes.
    For each changed script, notify all subscribed users via talk page.

    This is called from the `check_updates` view (GET /api/check-updates/)
    or can be called from a Toolforge cron job (future).

    Returns a summary dict for logging/display.
    """
    from .models import TrackedScript, UserPreference

    summary = {"checked": 0, "changed": 0, "notified": 0, "errors": 0}

    active_scripts = TrackedScript.objects.filter(is_active=True)

    for script in active_scripts:
        summary["checked"] += 1
        try:
            current_hash = _fetch_file_hash(script.source_repo, script.source_file)
        except Exception as e:
            logger.warning("Could not fetch hash for %s: %s", script.source_file, e)
            summary["errors"] += 1
            continue

        script.last_checked_at = django_timezone.now()

        if current_hash and current_hash != script.last_hash:
            script.has_update = True
            script.save(update_fields=["last_checked_at", "has_update"])
            summary["changed"] += 1

            # Notify the script owner if they are opted in
            pref = UserPreference.objects.filter(
                username=script.username, opted_out=False
            ).first()

            if pref:
                ok = _send_talk_page_notification(script)
                summary["notified" if ok else "errors"] += 1
        else:
            script.has_update = False
            script.save(update_fields=["last_checked_at", "has_update"])

    return summary


def _fetch_file_hash(repo_url: str, source_file: str) -> str | None:
    """
    Fetch the current content of a file from its repository and return a SHA-256 hash.
    Uses the same direct API approach as repo_fetcher.py.
    """
    from .repo_fetcher import fetch_files, RepoFetchError

    try:
        files = fetch_files(repo_url)
    except RepoFetchError:
        return None

    for f in files:
        if f["name"] == source_file:
            return hashlib.sha256(f["content"].encode("utf-8")).hexdigest()

    return None


def _send_talk_page_notification(script) -> bool:
    """
    Post a notification to User_talk:USERNAME on the target wiki.

    Uses the dedicated bot account (NOTIFIER_BOT_USERNAME / NOTIFIER_BOT_PASSWORD).
    Returns True if the edit succeeded, False otherwise.
    """
    from .models import NotificationLog

    if not BOT_USERNAME or not BOT_PASSWORD:
        logger.error(
            "NOTIFIER_BOT_USERNAME or NOTIFIER_BOT_PASSWORD not set in environment. "
            "Cannot send talk-page notifications."
        )
        return False

    talk_page = f"User talk:{script.username}"
    message   = _build_notification_message(script)

    log = NotificationLog(
        tracked_script=script,
        notified_username=script.username,
        notified_wiki=script.target_wiki,
        talk_page=talk_page,
    )

    try:
        bot = BotSession(script.target_wiki)
        bot.login(BOT_USERNAME, BOT_PASSWORD)

        result = bot.edit_page(
            page_title=talk_page,
            new_content=None,          # append, not replace (handled below)
            edit_summary=f"Script update notification: {script.source_file}",
        )
        # We override edit_page's replace mode by using the append parameter instead
        # Re-doing this properly:
        _append_to_talk_page(bot, talk_page, message, script.source_file)

        log.success = True
        log.save()
        logger.info("Notified %s on %s", script.username, script.target_wiki)
        return True

    except WikiAPIError as e:
        log.success = False
        log.error_message = str(e)
        log.save()
        logger.error("Failed to notify %s: %s", script.username, e)
        return False


def _append_to_talk_page(bot: "BotSession", talk_page: str, message: str, script_name: str):
    """
    Append a new section to a talk page instead of replacing the whole page.
    Uses MediaWiki's action=edit with section=new parameter.
    """
    csrf_token = bot._get_token("csrf")

    response = bot._post({
        "action": "edit",
        "title": talk_page,
        "section": "new",
        "sectiontitle": f"Script update available: {script_name}",
        "text": message,
        "summary": f"/* Script update available: {script_name} */ new section",
        "bot": "1",
        "token": csrf_token,
        "format": "json",
    })

    data = response.json()
    if "error" in data:
        raise WikiAPIError(data["error"].get("info", "Unknown error"))

    return data.get("edit", {})


def _build_notification_message(script) -> str:
    """
    Build the wikitext message to post on the user's talk page.
    Includes an opt-out link using a signed token (no login required to opt out).
    """
    from django.core import signing

    opt_out_token = signing.dumps(script.username, salt="notification-opt-out")
    opt_out_url   = f"{TOOLFORGE_URL}/notifications/opt-out/?token={opt_out_token}"

    return (
        f"Hello {{{{subst:PAGENAME}}}},\n\n"
        f"The source repository for '''[[{script.target_page}]]''' on "
        f"[[{script.target_wiki}|{script.target_wiki}]] has been updated.\n\n"
        f"; Source repository: [{script.source_repo} {script.source_repo}]\n"
        f"; File: <code>{script.source_file}</code>\n"
        f"; Target page: [[{script.target_page}]]\n\n"
        f"Please review the changes in the repository and update the wiki page "
        f"if you would like to deploy the new version. "
        f"You can use [[Toolforge:script-publisher|WikiScriptSync]] to help with this.\n\n"
        f"If you do not want to receive these notifications in future, "
        f"[{opt_out_url} click here to opt out].\n\n"
        f"— Sent by [[User:{BOT_USERNAME.split('@')[0]}|WikiScriptSync bot]] "
        f"([[Toolforge:script-publisher|tool]]) ~~~~"
    )


# ──────────────────────────────────────────────────────────────────────────────
# Track a new script after publish

def register_tracked_script(
    username: str,
    source_repo: str,
    source_file: str,
    target_wiki: str,
    target_page: str,
    initial_content: str,
) -> None:
    """
    Create or update a TrackedScript row after a successful publish.
    Stores the initial hash so we can detect future changes.
    """
    from .models import TrackedScript

    initial_hash = hashlib.sha256(initial_content.encode("utf-8")).hexdigest()

    TrackedScript.objects.update_or_create(
        username=username,
        source_repo=source_repo,
        source_file=source_file,
        target_wiki=target_wiki,
        target_page=target_page,
        defaults={
            "last_hash": initial_hash,
            "has_update": False,
            "is_active": True,
            "last_checked_at": django_timezone.now(),
        },
    )

def get_started(request):
    if get_current_user(request):
        return redirect("dashboard")

    if request.method == "POST" and request.POST.get("auth_type") == "botpassword":
        bot_username  = request.POST.get("bot_username", "").strip()
        bot_password  = request.POST.get("bot_password", "")
        wiki_username = request.POST.get("wiki_username", "").strip()

        if not bot_username or not bot_password or not wiki_username:
            return render(request, "publisher/get_started.html",
                          {"error": "All fields are required."})

        request.session["user_username"] = wiki_username
        request.session["user_groups"]   = []
        request.session["user_rights"]   = []
        request.session["user_token"]    = None
        request.session["bot_username"]  = bot_username
        request.session["bot_password"]  = bot_password
        request.session.set_expiry(86400)
        return redirect("dashboard")

    error = request.session.pop("auth_error", None)
    return render(request, "publisher/get_started.html", {"error": error})

@require_login
def notifications(request):
    """
    GET  — show notification subscription page
    POST — opt in or opt out
    """
    user = get_current_user(request)

    if request.method == "POST":
        action = request.POST.get("action")

        if action == "opt_in":
            pref, _ = UserPreference.objects.get_or_create(
                username=user["username"],
                defaults={"opted_out": False},
            )
            pref.opted_out    = False
            pref.opted_out_at = None
            pref.save()

        elif action == "opt_out":
            pref, _ = UserPreference.objects.get_or_create(username=user["username"])
            pref.opted_out    = True
            pref.opted_out_at = timezone.now()
            pref.save()

        return redirect("notifications")

    pref = UserPreference.objects.filter(username=user["username"]).first()
    tracked = TrackedScript.objects.filter(username=user["username"], is_active=True)
    notification_logs = NotificationLog.objects.filter(
        notified_username=user["username"]
    ).select_related("tracked_script")[:20]

    return render(request, "publisher/dashboard/notifications.html", {
        "current_user": user,
        "pref": pref,
        "tracked": tracked,
        "notification_logs": notification_logs,
        "notifier_bot_wiki": os.environ.get("NOTIFIER_BOT_WIKI", "meta.wikimedia.org"),
        "notifier_bot_display": os.environ.get("NOTIFIER_BOT_USERNAME", "").split("@")[0] or "WikiScriptSyncBot",
    })


def notification_opt_out(request):
    """
    Public opt-out endpoint. Reached via the link sent in a talk-page notification.
    No login required — uses a signed token to identify the user.
    """
    from django.core import signing
    from django.core.signing import BadSignature, SignatureExpired

    token = request.GET.get("token", "")
    try:
        username = signing.loads(token, salt="notification-opt-out", max_age=86400 * 30)
    except (BadSignature, SignatureExpired):
        return render(request, "publisher/opt_out_result.html", {
            "error": "This opt-out link is invalid or has expired. Please log in to manage your preferences.",
        })

    pref, _ = UserPreference.objects.get_or_create(username=username)
    pref.opted_out    = True
    pref.opted_out_at = timezone.now()
    pref.save()

    return render(request, "publisher/opt_out_result.html", {
        "username": username,
        "success": True,
    })

@csrf_exempt
@require_POST
def api_fetch_repo(request):
    if not get_current_user(request):
        return JsonResponse({"error": "Not authenticated."}, status=401)

    try:
        body = json.loads(request.body)
    except json.JSONDecodeError:
        return JsonResponse({"error": "Invalid JSON body."}, status=400)

    repo_url = body.get("repo_url", "").strip()
    if not repo_url:
        return JsonResponse({"error": "repo_url is required."}, status=400)

    try:
        files = fetch_files(repo_url)
    except RepoFetchError as e:
        return JsonResponse({"error": str(e)}, status=422)
    except Exception:
        logger.exception("Unexpected error in api_fetch_repo")
        return JsonResponse({"error": "Unexpected error fetching repository."}, status=500)

    return JsonResponse({"files": files, "count": len(files)})

@csrf_exempt
@require_POST
def api_publish(request):
    current_user = get_current_user(request)
    if not current_user:
        return JsonResponse({"error": "Not authenticated."}, status=401)

    try:
        body = json.loads(request.body)
    except json.JSONDecodeError:
        return JsonResponse({"error": "Invalid JSON body."}, status=400)

    method       = body.get("method", "")
    wiki         = body.get("wiki", "").strip()
    page         = body.get("page", "").strip()
    content      = body.get("content", "")
    edit_summary = body.get("edit_summary", "Published via WikiScriptSync").strip()
    source_repo  = body.get("source_repo", "").strip()
    source_file  = body.get("source_file", "").strip()
    version      = body.get("version", "").strip()

    if not wiki or not page or not content:
        return JsonResponse({"error": "wiki, page, and content are required."}, status=400)

    if method == "botpassword":
        return _publish_botpassword(
            request, current_user, body, wiki, page, content,
            edit_summary, source_repo, source_file, version,
        )
    elif method == "draft":
        return _publish_draft(
            current_user, wiki, page, content, edit_summary,
            source_repo, source_file, version,
        )
    else:
        return JsonResponse({"error": "method must be 'botpassword' or 'draft'."}, status=400)


def _publish_botpassword(
    request, current_user, body, wiki, page, content,
    edit_summary, source_repo, source_file, version,
):
    bot_username = body.get("bot_username", request.session.get("bot_username", "")).strip()
    bot_password = body.get("bot_password", request.session.get("bot_password", ""))

    if not bot_username or not bot_password:
        return JsonResponse(
            {"error": "bot_username and bot_password are required for BotPassword publish."},
            status=400,
        )

    log = PublishLog(
        username=current_user["username"],
        source_repo=source_repo, source_file=source_file,
        target_wiki=wiki, target_page=page,
        version=version, edit_summary=edit_summary,
        publish_method="botpassword", status="pending_review",
    )

    try:
        session = BotSession(wiki)
        session.login(bot_username, bot_password)
        result = session.edit_page(page, content, edit_summary)

        log.status = "success"
        log.save()

        # Register in tracking system so notifications work
        register_tracked_script(
            username=current_user["username"],
            source_repo=source_repo, source_file=source_file,
            target_wiki=wiki, target_page=page,
            initial_content=content,
        )

        return JsonResponse({
            "success": True, "method": "botpassword",
            "wiki": wiki, "page": page,
            "revision_id": result.get("newrevid"),
            "log_id": log.id,
        })

    except WikiAPIError as e:
        log.status = "failed"
        log.error_message = str(e)
        log.save()
        return JsonResponse({"error": str(e)}, status=422)

    except Exception:
        log.status = "failed"
        log.error_message = "Unexpected internal error."
        log.save()
        logger.exception("Unexpected error in BotPassword publish")
        return JsonResponse({"error": "Unexpected error during publishing."}, status=500)


def _publish_draft(
    current_user, wiki, page, content, edit_summary,
    source_repo, source_file, version,
):
    edit_url     = build_draft_edit_url(wiki, page, content)
    notification = build_talk_page_notification(
        wiki, page, source_repo, current_user["username"], edit_summary,
    )

    PublishLog.objects.create(
        username=current_user["username"],
        source_repo=source_repo, source_file=source_file,
        target_wiki=wiki, target_page=page,
        version=version, edit_summary=edit_summary,
        publish_method="draft", status="pending_review",
    )

    # Also register in tracking so the user gets notified of future changes
    register_tracked_script(
        username=current_user["username"],
        source_repo=source_repo, source_file=source_file,
        target_wiki=wiki, target_page=page,
        initial_content=content,
    )

    return JsonResponse({
        "success": True, "method": "draft",
        "edit_url": edit_url,
        "notification_wikitext": notification,
        "instructions": (
            "Open the edit URL to review and submit the change yourself. "
            "Copy the notification wikitext to the relevant talk page if needed."
        ),
    })


# ── API: check for upstream updates ──────────────────────────────────────────

@require_GET
def api_check_updates(request):
    """
    GET /api/check-updates/
    Checks all tracked scripts for changes and sends talk-page notifications.

    Protected: only staff or the tool admin should call this in production.
    On Toolforge this will be called by a cron job.
    For local testing, visit the URL while logged in.
    """
    if not get_current_user(request):
        return JsonResponse({"error": "Not authenticated."}, status=401)

    summary = check_and_notify_all()
    return JsonResponse({"status": "ok", "summary": summary})

==== DEPLOYMENT.md ====
# Script Publisher — Deployment Guide

## Local development setup

```bash
# 1. Clone the repository
git clone https://gitlab.wikimedia.org/toolforge-repos/script-publisher.git
cd script-publisher

# 2. Create and activate a virtual environment
python3 -m venv venv
source venv/bin/activate

# 3. Install dependencies
pip install -r requirements.txt

# 4. Copy the environment file and fill in your values
cp .env.example .env
# Edit .env — set OAUTH_REDIRECT_URI=http://localhost:8000/api/auth/callback

# 5. Run migrations
python manage.py migrate

# 6. Start the development server
python manage.py runserver
```

Open http://localhost:8000 in your browser.

---

## Environment variables

| Variable | Required | Description |
|---|---|---|
| `SECRET_KEY` | Yes (prod) | Django secret key. Generate with `python -c "import secrets; print(secrets.token_urlsafe(50))"` |
| `DEBUG` | No | Set to `True` for local dev, omit or `False` for production |
| `ALLOWED_HOSTS` | Yes (prod) | Comma-separated hostnames, e.g. `script-publisher.toolforge.org` |
| `OAUTH_CLIENT_ID` | Yes | Your OAuth consumer key |
| `OAUTH_CLIENT_SECRET` | Yes | Your OAuth consumer secret — never commit this |
| `OAUTH_REDIRECT_URI` | Yes | Must match exactly what is registered in your OAuth consumer. Use `http://localhost:8000/api/auth/callback` locally |
| `DATABASE_URL` | No | Leave unset to use SQLite locally. Set to a MariaDB URL on Toolforge |

---

## Toolforge deployment

### 1. Set environment variables via Toolforge secrets

```bash
# SSH into Toolforge
ssh login.toolforge.org

# Set each secret (do not put secrets in files that get committed)
toolforge envvars create SECRET_KEY "your-long-random-key"
toolforge envvars create OAUTH_CLIENT_SECRET "0534219c56e661de1ab794267912f6527b6a47a5"
toolforge envvars create OAUTH_CLIENT_ID "c4b5128cc24c0b2c46e4f63cca29de43"
toolforge envvars create OAUTH_REDIRECT_URI "https://script-publisher.toolforge.org/api/auth/callback"
toolforge envvars create DEBUG "False"
toolforge envvars create ALLOWED_HOSTS "script-publisher.toolforge.org"
```

### 2. Deploy using the build script

```bash
# From the Toolforge shell, inside the repo
bash build.sh
```

`build.sh` runs:
- `pip install -r requirements.txt`
- `python manage.py collectstatic --no-input`
- `python manage.py migrate`

### 3. OAuth callback URL verification

The callback URL registered in your OAuth consumer must be exactly:
```
https://script-publisher.toolforge.org/api/auth/callback
```

This is in `publisher/urls.py` as `path("api/auth/callback/", ...)`.
The trailing slash is handled by Django's `APPEND_SLASH` setting.

---

## Security checklist before production use

- [ ] `DEBUG=False` in production
- [ ] `SECRET_KEY` is unique and not committed to git
- [ ] `OAUTH_CLIENT_SECRET` is only in Toolforge environment variables
- [ ] `SESSION_COOKIE_SECURE=True` (automatic when `DEBUG=False`)
- [ ] `db.sqlite3` is in `.gitignore` (already is)
- [ ] `.env` is in `.gitignore` (already is)
- [ ] All dependencies are pinned in `requirements.txt`

## OAuth consumer registration

Your consumer is registered at:
https://meta.wikimedia.org/wiki/Special:OAuthConsumerRegistration/list

Current scope: `basic` (read username + groups only).

**Do not request JS-editing scopes** unless the community review at
`Steward_requests/Miscellaneous#OAuth_permissions` has been approved.
The current tool design (draft URL path + BotPassword path) does not require
JS-editing OAuth permissions.

## Troubleshooting

**`ModuleNotFoundError: No module named 'django'`**
→ Your virtual environment is not activated. Run `venv\Scripts\Activate.ps1` again.

**`django.db.OperationalError: no such table`**
→ Run `python manage.py migrate`

**OAuth callback gives 404**
→ Your OAUTH_REDIRECT_URI in .env doesn't match the server URL. For local dev, use `http://localhost:8000/api/auth/callback`.

**`DisallowedHost` error**
→ Add `127.0.0.1` and `localhost` to `ALLOWED_HOSTS` in your `.env`.
